# These are all the files that our program contains, along with their split variants if they have any
# You can modify the destination of the save location to specify where you want the new images to be saved
# run-script command is not included but can be run as well

# Load Images
#load skyline.png and call it 'skyline'
load ../res/skyline.jpg skyline

# Red Component
red-component skyline skylineRedComponent
red-component skyline skylineRedComponent split 50
save ../res/skylineRedComponent.png skylineRedComponent

# Green Component
green-component skyline skylineGreenComponent
green-component skyline skylineGreenComponent split 50
save ../res/skylineGreenComponent.png skylineGreenComponent


# Blue Component
blue-component skyline skylineBlueComponent
blue-component skyline skylineBlueComponent split 50
save ../res/skylineBlueComponent.png skylineBlueComponent

# Luma Component
luma-component skyline skylineLuma
luma-component skyline skylineLuma split 50
save ../res/skylineLuma.png skylineLuma

# Intensity Component
intensity-component skyline skylineIntensity
intensity-component skyline skylineIntensity split 50
save ../res/skylineIntensity.png skylineIntensity

# Value Component
value-component skyline skylineComponent
value-component skyline skylineComponent
save ../res/skylineComponent.png skylineComponent

# Horizontal Flip
horizontal-flip skyline skylineHorizontal
save ../res/skylineHorizontal.png skylineHorizontal

# Vertical Flip
vertical-flip skyline skylineVertical
save ../res/skylineVertical.png skylineVertical

# Brighten
brighten 100 skyline skylineBrighten
save ../res/skylineBrighten.png skylineBrighten

# Darken
brighten -100 skyline skylineDarker
save ../res/skylineDarker.png skylineDarker

# RGB Split
rgb-split skyline skylineRed skylineGreen skylineBlue
save ../res/skylineRed.png skylineRed
save ../res/skylineGreen.png skylineGreen
save ../res/skylineBlue.png skylineBlue

# RGB Combine
rgb-split skyline skylineRed skylineGreen skylineBlue
rgb-combine skylineCombine skylineRed skylineGreen skylineBlue
save ../res/skylineCombine.png skylineCombine

# Blur
blur skyline skylineBlur
blur skyline skylineBlur split 50
save ../res/skylineBlur.png skylineBlur

# Sharpen
sharpen skyline skylineSharpen
sharpen skyline skylineSharpen split 50
save ../res/skylineSharpen.png skylineSharpen

# Sepia
sepia skyline skylineSepia
sepia skyline skylineSepia split 50
save ../res/skylineSepia.png skylineSepia


# Color-Correct
color-correct skyline skylineColorCorrection split 50
color-correct skyline skylineColorCorrection
save ../res/skylineColorCorrection.png skylineColorCorrection

# Levels-Adjust
levels-adjust 0 99 255 skyline skylineLevelsAdjust1
save ../res/skylineLevelsAdjust1.png skylineLevelsAdjust1
levels-adjust 0 10 100 skyline skylineLevelsAdjust2
save ../res/skylineLevelsAdjust2.png skylineLevelsAdjust2
levels-adjust 0 90 255 skyline skylineLevelsAdjust3 split 50
save ../res/skylineLevelsAdjust3.png skylineLevelsAdjust3

# Compress
compress 90 skyline skylineCompressed90
save ../res/skylineCompressed90.png skylineCompressed90
compress 99 skyline skylineCompressed99
save ../res/skylineCompressed99.png skylineCompressed99

# Histogram
histogram skylineCompressed90 skylineCompressed90Histogram
save ../res/skylineCompressed90Histogram.png skylineCompressed90Histogram

histogram skylineCompressed99 skylineCompressed99Histogram
save ../res/skylineCompressed99Histogram.png skylineCompressed99Histogram

histogram colorCorrection skylineColorCorrection
save ../res/skylineColorCorrection.png skylineColorCorrection

histogram skylineLevelsAdjust1 skylineLevelsAdjust1Histogram
save ../res/skylineLevelsAdjust1Histogram.png skylineLevelsAdjust1Histogram

histogram skylineLevelsAdjust2 skylineLevelsAdjust2Histogram
save ../res/skylineLevelsAdjust2Histogram.png skylineLevelsAdjust2Histogram

# Save commands have been used throughout the file so we won't include a section for it here
# Exit Program, only used for shell
# exit
